## Question

### **나와 가장 까가운 좌표 구하기**

<br/>

## Solve

```java
import java.io.*;
import java.util.*;

class Dot {
  int x;
  int y;

  double distance; // 내 좌표와의 거리

  public Dot(int x, int y, double distance) {
    this.x = x;
    this.y = y;
    this.distance = distance;
  }
}

public class Mission1_SurpriseAsg_02 {
  public static void main(String[] args) throws IOException {
    BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
    StringTokenizer st = new StringTokenizer(br.readLine(), " ");

    int myX = Integer.parseInt(st.nextToken());
    int myY = Integer.parseInt(st.nextToken());

    ArrayList<Dot> list = new ArrayList<>();

    double min = Double.MAX_VALUE; // 가장 가까운 거리

    while (list.size() < 10) {
      st = new StringTokenizer(br.readLine(), " ");
      int x = Integer.parseInt(st.nextToken());
      int y = Integer.parseInt(st.nextToken());

      if (myX != x && myY != y) {
        double distance = getDistance(myX, myY, x, y);
        list.add(new Dot(x, y, distance));

        min = Math.min(min, distance);
      }
    }

    // 가장 가까운 거리의 좌표 모두 출력
    for (int i = 0; i < list.size(); i++) {
      Dot dot = list.get(i);

      if (dot.distance == min) {
        System.out.printf("(%d, %d) ", dot.x, dot.y);
      }
    }
  }

  // 내 좌표와의 거리 구하기
  public static double getDistance(int myX, int myY, int x, int y) {
    double dX = Math.pow(myX - x, 2);
    double dY = Math.pow(myY - y, 2);

    return Math.sqrt(dX+dY);
  }
}
```
